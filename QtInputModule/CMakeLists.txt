file (GLOB H_FILES *.h)
file (GLOB MOC_H_FILES QtInputService.h QtInputModule.h KeyBindingsConfigWindow.h)

use_package (BOOST)
use_package (POCO)
use_package (OGRE)
use_package (QT4)

link_package (BOOST)
link_package (POCO)
link_package (OGRE)
link_package (QT4)
link_modules (Core Foundation Interfaces)
